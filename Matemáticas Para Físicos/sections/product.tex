\section*{Properties of Products}
\begin{align*}
    & \prod_{i=1}^n c = c^n \;\;\;\; \prod_{i=1}^n (k \cdot a_i) = k^n \prod_{i=1}^n a_i \;\;\;\; \prod_{i=m}^n (a_i \cdot b_i) = \left( \prod_{i=m}^n a_i \right) \left( \prod_{i=m}^n b_i \right) \\
    & \prod_{i=m}^n \frac{a_i}{b_i} = \frac{\prod_{i=m}^n a_i}{\prod_{i=m}^n b_i} \;\;\;\; \ln \left( \prod_{i=m}^n a_i \right) = \sum_{i=m}^n \ln(a_i) \;\;\;\; \prod_{i=m}^n \frac{a_i}{a_{i-1}} = \frac{a_n}{a_{m-1}} \\
    & \prod_{i=m}^n a_i = \prod_{j=m+k}^{n+k} a_{j-k} \;\;\;\; \prod_{i=1}^n i = n! \;\;\;\; \prod_{i=1}^\infty (1+a_i) < \infty \iff \sum_{i=1}^\infty a_i < \infty
\end{align*}
