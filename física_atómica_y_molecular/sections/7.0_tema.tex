\section*{Física molecular: estructura electrónica}
\begin{align*}
&\underline{\text{Hamiltoniano de moléculas diatómicas}}: H=T_N +T_e+U_{ee}+ U_{NN} + U_{eN}\\
&T_N = -\frac{\hbar^2}{2\mu}\nabla_{R}^2\; \; \; \; \mu=\frac{M_A M_B}{M_A+M_B}\;\;\;\;\;\; T_e=-\sum_{i=1}^N\frac{\hbar^2}{2m}\nabla_{\vec{r}_i}^2\;\;\;\;U_{ee}=\sum_{\underset{i>j}{i,j=1}}^N\frac{e^2}{4\pi\varepsilon_0 |\vec{r}_i-\vec{r}_j|}
\\[-5pt]
&U_{NN}=\frac{Z_AZ_B e^2}{4\pi \varepsilon_0|\vec{R}|}\;\;\;\;U_{eN}=\!-\!\sum_{i=1}^N\frac{e^2}{4\pi \varepsilon_0}\left[\frac{Z_A}{|\vec{r}_i\!-\! \vec{R}_A|}\!+\!\frac{Z_B}{|\vec{r}_i\!-\! \vec{R}_B|}\right]\;\;\; \begin{array}{l}
     &H^e \!=\! T_e\!+\!U_{ee}\!+\!U_{eN}  \\
     &H^N \!=\! T_N \!+\! U_{NN}
\end{array}
\\
&\text{Núcleos $A,B$ ($M_A, M_B$), $N$ e$^-$ ($m$), SR $\sim$ C.M. moléc $\vec{R}=\vec{R}_B\! -\! \vec{R}_A$, $\vec{r}_k\!\equiv\!$ pos e$\!^- \;k$-ésim}\\&
    \text{Aprox. adiabática: }\psi(\vec{R}, \vec{r}_i)= \phi_q^e(r\vec{r}_i, \vec{R}^*)\;\phi_q ^N(\vec{R})\; \;\;\; 
    \begin{array}{c}\vspace{-5pt}
         &   \text{$\vec{R}^*\equiv$ pos. núc. fija}\vspace{-5pt} \\
         & \text{$q=1,2,...$($q=1$ nivel fund)}
    \end{array}\\
    &H^e\phi_q^e (\vec{r}_i, \vec{R}^*)=E_q^e\phi_q^e (\vec{r}_i, \vec{R}^*)\;\;\;\; (T_N+U_q^{ef})\phi_q^N(\vec{R})=E_q \phi_q^N(\vec{R})\;\;\;\; U_q^{ef}=U_{NN}+E_q^e
\end{align*}
