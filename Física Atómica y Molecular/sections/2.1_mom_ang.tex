\section*{Momento angular propiedades\normalfont{ ($m$ proyec. mom. ang. $j$) \;\; \;  $\hat{J}=\hat{\vec{J}}=\hat{J}_x+\hat{J}_y+\hat{J}_z$}}
\begin{align*}
    &\forall \text{ momento angular $\vec{J}$ t.q. } [\hat{J}^2,H]=[\hat{J}_z, H]=0 : \:\;\; [\hat{J}_i, \hat{J}_k]=i\hbar \;\epsilon_{ikl} \ \, \hat{J}_l\\
    &\hat{J}_{\pm} = \hat{J}_x\pm i \hat{J}_y\;\; \: \; \hat{J}^2=\frac{1}{2}[\hat{J}_+\hat{J}_- + \hat{J}_-\hat{J}_+]+\hat{J}_z^2 \; \; \; [\hat{J}_z, \hat{\pm}]=\pm \hbar \hat{J}_{\pm}\; \; \, [\hat{J}_+,\hat{J}_-]=2\hbar \hat{J}_z\\
    &\scalebox{0.95}{$\hat{J}^2|j \; m\rangle = \hbar^2 j(j+1)|j \;m \rangle \; \; \; \hat{J}_z |j \; m\rangle = m\hbar  |j \; m\rangle\; \; \hat{J}_{\pm}|j\; m\rangle = \hbar\sqrt{j(j+1)-m(m\pm1)}|j\;m\!\pm\!1\rangle$}\\
    &\hat{\vec{j}} = \hat{\vec{l}} + \hat{\vec{s}}=l\oplus s= \{|l-s|, |l-s|+1, ..., l+s\} \text{ (pasos de 1)} \;\;\;\;
\hat{\vec{j}}^{\,2} = \hat{\vec{l}}^{\,2} + \hat{\vec{s}}^{\,2} + 2\,\hat{\vec{l}}\cdot\hat{\vec{s}}
\end{align*}
